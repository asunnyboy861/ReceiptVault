### 6.2 OCR服务（Vision Framework）

```swift
import Vision
import UIKit

actor OCRService {
    static let shared = OCRService()

    struct OCRResult {
        var fullText: String = ""
        var merchant: String?
        var amount: Double?
        var date: Date?
        var address: String?
        var phone: String?
        var currency: String = "USD"
        var lineItems: [LineItem] = []
        var confidence: Float = 0
    }

    struct LineItem {
        let name: String
        let quantity: Int
        let unitPrice: Double
    }

    func recognize(image: UIImage) async throws -> OCRResult {
        guard let cgImage = image.cgImage else {
            throw OCRError.invalidImage
        }

        let request = VNRecognizeTextRequest()
        request.recognitionLevel = .accurate
        request.recognitionLanguages = ["en-US", "en-GB"]
        request.usesLanguageCorrection = true

        let observations = try await withCheckedThrowingContinuation {
            (continuation: CheckedContinuation<[VNRecognizedTextObservation], Error>) in
            let handler = VNImageRequestHandler(cgImage: cgImage, options: [:])
            request.completionHandler = { request, error in
                if let error = error {
                    continuation.resume(throwing: error)
                    return
                }
                let obs = request.results as? [VNRecognizedTextObservation] ?? []
                continuation.resume(returning: obs)
            }
            try? handler.perform([request])
        }

        var result = OCRResult()
        var lines: [(text: String, confidence: Float)] = []

        for observation in observations {
            guard let candidate = observation.topCandidates(1).first else { continue }
            lines.append((text: candidate.string, confidence: candidate.confidence))
            result.fullText += candidate.string + "\n"
        }

        result.confidence = lines.isEmpty ? 0 : lines.map(\.confidence).reduce(0, +) / Float(lines.count)
        result.merchant = extractMerchant(from: lines)
        result.amount = extractAmount(from: lines)
        result.date = extractDate(from: lines)
        result.address = extractAddress(from: lines)
        result.phone = extractPhone(from: lines)
        result.currency = detectCurrency(from: lines)
        result.lineItems = extractLineItems(from: lines)

        return result
    }

    private func extractMerchant(from lines: [(text: String, confidence: Float)]) -> String? {
        let knownMerchants = [
            "walmart": "Walmart", "target": "Target", "costco": "Costco",
            "amazon": "Amazon", "starbucks": "Starbucks", "mcdonalds": "McDonald's",
            "whole foods": "Whole Foods", "trader joe": "Trader Joe's",
            "home depot": "The Home Depot", "lowes": "Lowe's",
            "cvs": "CVS Pharmacy", "walgreens": "Walgreens",
            "shell": "Shell", "chevron": "Chevron", "bp": "BP"
        ]

        for line in lines.prefix(5) {
            let lower = line.text.lowercased()
            for (key, value) in knownMerchants {
                if lower.contains(key) { return value }
            }
        }

        for line in lines.prefix(3) {
            let trimmed = line.text.trimmingCharacters(in: .whitespaces)
            if trimmed.count >= 2 && trimmed.count <= 50 && !trimmed.allSatisfy({ $0.isNumber }) {
                return trimmed
            }
        }
        return nil
    }

    private func extractAmount(from lines: [(text: String, confidence: Float)]) -> Double? {
        let totalPatterns = [
            /TOTAL\s*\$?([\d,]+\.\d{2})/i,
            /AMOUNT\s*DUE\s*\$?([\d,]+\.\d{2})/i,
            /BALANCE\s*DUE\s*\$?([\d,]+\.\d{2})/i
        ]

        for line in lines {
            for pattern in totalPatterns {
                if let match = line.text.firstMatch(of: pattern),
                   let amount = Double(match.1.replacingOccurrences(of: ",", with: "")) {
                    return amount
                }
            }
        }

        var maxAmount: Double = 0
        let amountPattern = /\$?([\d,]+\.\d{2})/
        for line in lines {
            if let match = line.text.firstMatch(of: amountPattern),
               let amount = Double(match.1.replacingOccurrences(of: ",", with: "")) {
                maxAmount = max(maxAmount, amount)
            }
        }
        return maxAmount > 0 ? maxAmount : nil
    }

    private func extractDate(from lines: [(text: String, confidence: Float)]) -> Date? {
        let formatter = DateFormatter()
        formatter.locale = Locale(identifier: "en_US")
        for line in lines {
            for fmt in ["MM/dd/yyyy", "yyyy-MM-dd", "MMM dd, yyyy", "MM-dd-yyyy"] {
                formatter.dateFormat = fmt
                if let date = formatter.date(from: line.text.trimmingCharacters(in: .whitespaces)) {
                    return date
                }
            }
            let datePattern = /(\d{1,2}\/\d{1,2}\/\d{4})/
            if let match = line.text.firstMatch(of: datePattern) {
                formatter.dateFormat = "MM/dd/yyyy"
                if let date = formatter.date(from: String(match.1)) {
                    return date
                }
            }
        }
        return nil
    }

    private func extractAddress(from lines: [(text: String, confidence: Float)]) -> String? {
        let addressPattern = /\d+\s+\w+\s+(ST|AVE|BLVD|DR|RD|LN|WAY|CT|PL|CIRCLE).*/i
        for line in lines {
            if let match = line.text.firstMatch(of: addressPattern) {
                return String(match.output)
            }
        }
        return nil
    }

    private func extractPhone(from lines: [(text: String, confidence: Float)]) -> String? {
        let phonePattern = /(\(?\d{3}\)?[-.\s]?\d{3}[-.\s]?\d{4})/
        for line in lines {
            if let match = line.text.firstMatch(of: phonePattern) {
                return String(match.1)
            }
        }
        return nil
    }

    private func detectCurrency(from lines: [(text: String, confidence: Float)]) -> String {
        let fullText = lines.map(\.text).joined(separator: " ")
        if fullText.contains("\u{20AC}") { return "EUR" }
        if fullText.contains("\u{00A3}") { return "GBP" }
        if fullText.contains("\u{00A5}") { return "JPY" }
        if fullText.contains("CA$") { return "CAD" }
        return "USD"
    }

    private func extractLineItems(from lines: [(text: String, confidence: Float)]) -> [LineItem] {
        var items: [LineItem] = []
        let itemPattern = /^(.+?)\s+(\d+x\s*)?\$?([\d,]+\.\d{2})$/

        var inItemSection = false
        for line in lines {
            let text = line.text.trimmingCharacters(in: .whitespaces)
            if text.uppercased().contains("SUBTOTAL") { inItemSection = false; break }
            if text.contains("---") { inItemSection = true; continue }
            if !inItemSection { continue }

            if let match = text.firstMatch(of: itemPattern) {
                let name = String(match.1).trimmingCharacters(in: .whitespaces)
                let priceStr = String(match.3).replacingOccurrences(of: ",", with: "")
                guard let price = Double(priceStr) else { continue }
                items.append(LineItem(name: name, quantity: 1, unitPrice: price))
            }
        }
        return items
    }
}

enum OCRError: LocalizedError {
    case invalidImage
    case noTextFound
    case recognitionFailed

    var errorDescription: String? {
        switch self {
        case .invalidImage: "Could not process this image"
        case .noTextFound: "No text found in this image"
        case .recognitionFailed: "Text recognition failed"
        }
    }
}
```

### 6.4 CSV导出服务

```swift
import Foundation

struct CSVExporter {
    static func export(receipts: [ReceiptModel], dateRange: DateRange? = nil) -> URL? {
        let filtered = applyDateFilter(receipts, range: dateRange)
        guard !filtered.isEmpty else { return nil }

        var csv = "Date,Merchant,Category,Amount,Currency,Address,Phone,Notes\n"

        let formatter = DateFormatter()
        formatter.dateFormat = "yyyy-MM-dd"

        for receipt in filtered {
            let date = formatter.string(from: receipt.date)
            let merchant = escapeCSV(receipt.merchant)
            let category = receipt.category
            let amount = String(format: "%.2f", receipt.amount)
            let currency = receipt.currency
            let address = escapeCSV(receipt.address)
            let phone = escapeCSV(receipt.phone)
            let notes = escapeCSV(receipt.notes)

            csv += "\(date),\(merchant),\(category),\(amount),\(currency),\(address),\(phone),\(notes)\n"
        }

        let tempDir = FileManager.default.temporaryDirectory
        let fileName = "ReceiptVault_Export_\(formatter.string(from: Date())).csv"
        let fileURL = tempDir.appendingPathComponent(fileName)

        do {
            try csv.write(to: fileURL, atomically: true, encoding: .utf8)
            return fileURL
        } catch {
            return nil
        }
    }

    private static func escapeCSV(_ field: String) -> String {
        if field.contains(",") || field.contains("\"") || field.contains("\n") {
            return "\"\(field.replacingOccurrences(of: "\"", with: "\"\""))\""
        }
        return field
    }

    private static func applyDateFilter(_ receipts: [ReceiptModel], range: DateRange?) -> [ReceiptModel] {
        guard let range = range else { return receipts }
        return receipts.filter { receipt in
            receipt.date >= range.start && receipt.date <= range.end
        }
    }

struct DateRange {
    let start: Date
    let end: Date

    static var thisMonth: DateRange {
        let calendar = Calendar.current
        let now = Date()
        let start = calendar.date(from: calendar.dateComponents([.year, .month], from: now))!
        let end = calendar.date(byAdding: .month, value: 1, to: start)!
        return DateRange(start: start, end: end)
    }

    static var thisYear: DateRange {
        let calendar = Calendar.current
        let now = Date()
        let start = calendar.date(from: calendar.dateComponents([.year], from: now))!
        let end = calendar.date(byAdding: .year, value: 1, to: start)!
        return DateRange(start: start, end: end)
    }
}
```

```swift
import SwiftUI
import SwiftData

struct ContentView: View {
    @Environment(\.modelContext) private var modelContext
    @Query(sort: \ReceiptModel.date, order: .reverse) private var receipts: [ReceiptModel]
    @State private var showScanner = false
    @State private var selectedCategory: String? = nil
    @State private var searchText = ""

    var filteredReceipts: [ReceiptModel] {
        receipts.filter { receipt in
            let matchSearch = searchText.isEmpty ||
                receipt.merchant.localizedCaseInsensitiveContains(searchText) ||
                receipt.notes.localizedCaseInsensitiveContains(searchText)
            let matchCategory = selectedCategory == nil || receipt.category == selectedCategory
            return matchSearch && matchCategory
        }
    }

    var monthlyTotal: String {
        let calendar = Calendar.current
        let thisMonth = receipts.filter { calendar.isDate($0.date, equalTo: .now, toGranularity: .month) }
        let total = thisMonth.reduce(0) { $0 + $1.amount }
        return String(format: "$%.2f", total)
    }

    var body: some View {
        NavigationStack {
            VStack(spacing: 0) {
                monthlySummaryCard
                CategoryFilterBar(selectedCategory: $selectedCategory)
                receiptList
            }
            .navigationTitle("ReceiptVault")
            .searchable(text: $searchText, prompt: "Search receipts...")
            .toolbar {
                ToolbarItem(placement: .topBarTrailing) {
                    NavigationLink { StatsView() } label: {
                        Image(systemName: "chart.pie")
                    }
                }
            }
            .sheet(isPresented: $showScanner) {
                ScannerSheet()
            }
            .overlay(alignment: .bottom) {
                scanButton
            }
        }
    }

    private var monthlySummaryCard: some View {
        VStack(spacing: 4) {
            Text("This Month")
                .font(.subheadline)
                .foregroundStyle(.secondary)
            Text(monthlyTotal)
                .font(.system(size: 36, weight: .bold, design: .rounded))
                .contentTransition(.numericText())
            Text("\(receipts.count) receipts")
                .font(.caption)
                .foregroundStyle(.secondary)
        }
        .frame(maxWidth: .infinity)
        .padding(.vertical, 16)
        .background(Color(.systemBackground))
    }

    private var scanButton: some View {
        Button { showScanner = true } label: {
            HStack(spacing: 8) {
                Image(systemName: "camera.viewfinder")
                Text("Scan Receipt")
            }
            .font(.headline)
            .foregroundStyle(.white)
            .padding(.horizontal, 32)
            .padding(.vertical, 16)
            .background(Color.accentColor, in: Capsule())
        }
        .padding(.bottom, 16)
    }

    private var receiptList: some View {
        List {
            ForEach(filteredReceipts) { receipt in
                NavigationLink {
                    ReceiptDetailView(receipt: receipt)
                } label: {
                    ReceiptRow(receipt: receipt)
                }
            }
            .onDelete(perform: deleteReceipts)
        }
        .listStyle(.plain)
        .padding(.bottom, 80)
    }

    private func deleteReceipts(at offsets: IndexSet) {
        for index in offsets {
            modelContext.delete(filteredReceipts[index])
        }
    }
}
```
